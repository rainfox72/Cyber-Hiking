```typescript
// ── Background Music ──

startMusic(): void {
  if (this.musicElement) return;
  this.musicElement = new Audio("/music/Witch Parade Assassin.mp3");
  this.musicElement.loop = true;
  this.musicElement.volume = 0; // Start silent for fade-in
  this.musicElement.play().catch(() => {/* Autoplay blocked — will retry on interaction */});

  // 3-second fade-in
  const targetVolume = this.muted ? 0 : 0.3;
  const fadeStep = targetVolume / 60; // 60 steps over ~3s
  let currentVol = 0;
  const fadeInterval = setInterval(() => {
    currentVol += fadeStep;
    if (currentVol >= targetVolume) {
      currentVol = targetVolume;
      clearInterval(fadeInterval);
    }
    if (this.musicElement) this.musicElement.volume = currentVol;
  }, 50);
}

stopMusic(): void {
  if (this.musicElement) {
    this.musicElement.pause();
    this.musicElement = null;
  }
}
```

```typescript
export function HumanMarker({ x, y, healthPercent, lastAction }: HumanMarkerProps) {
  const color =
    healthPercent > 60 ? "var(--neon-green)" :
    healthPercent > 30 ? "var(--amber)" :
    "var(--danger)";

  const pose = actionToPose(lastAction);

  return (
    <g
      className="human-marker"
      transform={`translate(${x}, ${y})`}
      style={{ transition: "transform 1.2s ease-in-out" }}
    >
      {pose === "idle" && <IdlePose color={color} />}
      {pose === "walking" && <WalkingPose color={color} />}
      {pose === "camping" && <CampingPose color={color} />}
      {pose === "eating" && <EatingPose color={color} />}
      {pose === "drinking" && <DrinkingPose color={color} />}
      {pose === "resting" && <RestingPose color={color} />}
      {pose === "mapping" && <MappingPose color={color} />}
      {pose === "medicine" && <MedicinePose color={color} />}
      {pose === "waiting" && <WaitingPose color={color} />}

      {/* Pulsing glow */}
      <circle cx="0" cy="-3" r="8" fill="none" stroke={color} strokeWidth="0.5" opacity="0.3">
        <animate attributeName="r" values="8;12;8" dur="2s" repeatCount="indefinite" />
        <animate attributeName="opacity" values="0.3;0.1;0.3" dur="2s" repeatCount="indefinite" />
      </circle>
    </g>
  );
}
```
