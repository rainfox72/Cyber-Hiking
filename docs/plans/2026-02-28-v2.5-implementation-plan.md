```typescript
/**
 * Ollama-powered AI decision service for auto-play mode.
 * Sends game context to Ollama and gets back a chosen action + reasoning.
 */

import type { GameAction, GameState, Waypoint } from "../engine/types.ts";
import { WAYPOINTS } from "../data/waypoints.ts";

const OLLAMA_URL = "http://localhost:11434/api/generate";
const MODEL = "llama3.1:8b";
const DECISION_TIMEOUT_MS = 15000;

export interface AIDecision {
  action: GameAction;
  reasoning: string;
}

/**
 * Build a decision prompt from the current game state and valid actions.
 */
function buildDecisionPrompt(state: GameState, validActions: GameAction[]): string {
  const wp = WAYPOINTS[state.player.currentWaypointIndex];
  const { weather, time, player } = state;

  return `You are an AI mountaineer on the Ao Tai Line (鳌太线), a dangerous 80km ridge traverse in China's Qinling Mountains. Choose the best action to survive and reach the summit.

Current state:
- Location: ${wp.name} (${wp.nameCN}), ${wp.elevation}m, terrain: ${wp.terrain.replace("_", " ")}
- Weather: ${weather.current}, intensity ${Math.round(weather.intensity * 100)}%, wind ${weather.windSpeed}km/h
- Time: Day ${time.day}, ${String(Math.floor(time.hour)).padStart(2, "0")}:00 (${time.timeOfDay})
- Energy: ${Math.round(player.energy)}%, Hydration: ${Math.round(player.hydration)}%, Body Temp: ${Math.round(player.bodyTemp)}%
- O2 Sat: ${Math.round(player.o2Saturation)}%, Morale: ${Math.round(player.morale)}%
- Food: ${player.food} meals, Water: ${player.water.toFixed(1)}L, Gear: ${Math.round(player.gear)}%, Medicine: ${player.medicine}
- Lost: ${player.isLost}
- Turn: ${state.turnNumber}

Available actions: ${validActions.join(", ")}

Respond ONLY with valid JSON (no markdown, no explanation outside the JSON):
{"action": "one_of_the_available_actions", "reasoning": "one sentence explaining why"}`;
}

/**
 * Parse Ollama's response into an AIDecision.
 * Falls back to null if the response is malformed.
 */
function parseDecisionResponse(text: string, validActions: GameAction[]): AIDecision | null {
  try {
    // Try to extract JSON from the response (Ollama sometimes wraps in markdown)
    const jsonMatch = text.match(/\{[^}]*"action"[^}]*\}/);
    if (!jsonMatch) return null;

    const parsed = JSON.parse(jsonMatch[0]);
    const action = parsed.action as GameAction;
    const reasoning = parsed.reasoning ?? "No reasoning provided.";

    // Validate the action is in the valid list
    if (!validActions.includes(action)) return null;

    return { action, reasoning };
  } catch {
    return null;
  }
}

/**
 * Ask Ollama to decide the next action.
 * Returns null if Ollama is unavailable, times out, or returns invalid response.
 */
export async function generateDecision(
  state: GameState,
  validActions: GameAction[],
): Promise<AIDecision | null> {
  if (validActions.length === 0) return null;

  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), DECISION_TIMEOUT_MS);

  try {
    const response = await fetch(OLLAMA_URL, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        model: MODEL,
        prompt: buildDecisionPrompt(state, validActions),
        stream: false,
        options: {
          temperature: 0.7,
          num_predict: 60,
          top_p: 0.9,
        },
      }),
      signal: controller.signal,
    });

    if (!response.ok) return null;

    const data = await response.json();
    const text = data.response?.trim();
    if (!text) return null;

    return parseDecisionResponse(text, validActions);
  } catch {
    return null;
  } finally {
    clearTimeout(timeoutId);
  }
}

/**
 * Heuristic fallback bot when Ollama is unavailable.
 * Simple survival logic: address critical vitals first, then push forward.
 */
export function heuristicDecision(
  state: GameState,
  validActions: GameAction[],
): AIDecision {
  const p = state.player;

  // Critical vitals — address immediately
  if (p.energy < 30 && p.food > 0 && validActions.includes("eat")) {
    return { action: "eat", reasoning: "Energy critically low — eating to survive." };
  }
  if (p.hydration < 30 && p.water > 0 && validActions.includes("drink")) {
    return { action: "drink", reasoning: "Dehydration imminent — drinking water." };
  }
  if (p.bodyTemp < 30 && validActions.includes("set_camp")) {
    return { action: "set_camp", reasoning: "Hypothermia risk — making camp to warm up." };
  }
  if (p.o2Saturation < 25 && p.medicine > 0 && validActions.includes("use_medicine")) {
    return { action: "use_medicine", reasoning: "Severe altitude sickness — taking medicine." };
  }

  // Moderate vitals — rest or camp
  if (p.energy < 40 && validActions.includes("set_camp")) {
    return { action: "set_camp", reasoning: "Low energy — camping to recover." };
  }
  if (p.energy < 40 && validActions.includes("rest")) {
    return { action: "rest", reasoning: "Low energy — resting briefly." };
  }

  // Lost — check map
  if (p.isLost && validActions.includes("check_map")) {
    return { action: "check_map", reasoning: "Lost — checking map to find the trail." };
  }

  // Default: push forward
  if (validActions.includes("push_forward")) {
    return { action: "push_forward", reasoning: "Conditions acceptable — pressing onward." };
  }

  // Fallback: first valid action
  return { action: validActions[0], reasoning: "Taking the only available action." };
}
```

```typescript

  toggleAutoPlay: () => {
    const store = get();
    if (store.autoPlayEnabled) {
      // Turning off
      set({ autoPlayEnabled: false, isAIThinking: false });
      return;
    }

    // Turning on
    set({ autoPlayEnabled: true });

    const runLoop = async () => {
      // Wait initial delay
      await new Promise((r) => setTimeout(r, 2000));

      while (true) {
        const current = get();
        if (!current.autoPlayEnabled || current.gamePhase !== "playing" || current.isProcessing) {
          set({ isAIThinking: false });
          break;
        }

        // Get valid actions
        const currentState = gameStateFromStore(current);
        const allActions: GameAction[] = [
          "push_forward", "set_camp", "descend", "check_map",
          "rest", "eat", "drink", "use_medicine",
        ];
        const validActions = allActions.filter((a) =>
          validateAction(currentState, a, WAYPOINTS),
        );

        if (validActions.length === 0) break;

        set({ isAIThinking: true });

        // Try Ollama decision, fall back to heuristic
        let decision = current.ollamaConnected
          ? await generateDecision(currentState, validActions)
          : null;

        if (!decision) {
          decision = heuristicDecision(currentState, validActions);
        }

        // Check if still enabled after async call
        if (!get().autoPlayEnabled) {
          set({ isAIThinking: false });
          break;
        }

        // Log AI reasoning
        const aiEntry: LogEntry = {
          turnNumber: currentState.turnNumber + 1,
          text: `[AI] ${decision.reasoning}`,
          type: "ai" as LogEntry["type"],
          timestamp: `Day ${currentState.time.day}, ${String(Math.floor(currentState.time.hour)).padStart(2, "0")}:00`,
        };
        set((s) => ({ log: [...s.log, aiEntry], isAIThinking: false }));

        // Perform the action
        get().performAction(decision.action);

        // Wait between turns
        await new Promise((r) => setTimeout(r, 3500));
      }
    };

    runLoop();
  },

  stopAutoPlay: () => {
    set({ autoPlayEnabled: false, isAIThinking: false });
  },
```
